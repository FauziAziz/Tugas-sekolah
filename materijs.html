1. Pengenalan JavaScript

JavaScript adalah bahasa pemrograman yang digunakan untuk membuat website interaktif dan dinamis.

JavaScript berjalan di browser (client-side) dan bisa juga di server (Node.js).

2. Cara Menulis JavaScript

Internal JavaScript:

<script>
    console.log("Hello World!");
</script>

External JavaScript:

<script src="script.js"></script>

Inline JavaScript:

<button onclick="alert('Hello!')">Klik Saya</button>

6. Operator

Operator Aritmatika:

let a = 10, b = 3;

console.log(a + b);  // 13 (penjumlahan)
console.log(a - b);  // 7 (pengurangan)
console.log(a * b);  // 30 (perkalian)
console.log(a / b);  // 3.333 (pembagian)
console.log(a % b);  // 1 (modulus/sisa bagi)
console.log(a ** b); // 1000 (pangkat)

// Increment & Decrement
let x = 5;
x++;  // x = 6
x--;  // x = 5

Operator Assignment:

let x = 10;
x += 5;  // x = x + 5  → 15
x -= 3;  // x = x - 3  → 12
x *= 2;  // x = x * 2  → 24
x /= 4;  // x = x / 4  → 6
x %= 4;  // x = x % 4  → 2

Operator Perbandingan:

console.log(5 == "5");   // true (nilai sama)
console.log(5 === "5");  // false (nilai dan tipe harus sama)
console.log(5 != "5");   // false
console.log(5 !== "5");  // true
console.log(5 > 3);      // true
console.log(5 < 3);      // false
console.log(5 >= 5);     // true
console.log(5 <= 3);     // false

Operator Logika:

let x = true, y = false;

console.log(x && y);  // false (AND)
console.log(x || y);  // true (OR)
console.log(!x);      // false (NOT)

8. Percabangan (Conditional)

If-Else:

let nilai = 75;

if (nilai >= 80) {
    console.log("A");
} else if (nilai >= 70) {
    console.log("B");
} else if (nilai >= 60) {
    console.log("C");
} else {
    console.log("D");
}

Ternary Operator:

let umur = 20;
let status = (umur >= 17) ? "Dewasa" : "Anak-anak";
console.log(status); // "Dewasa"

Switch:

let hari = 3;

switch (hari) {
    case 1:
        console.log("Senin");
        break;
    case 2:
        console.log("Selasa");
        break;
    case 3:
        console.log("Rabu");
        break;
    default:
        console.log("Hari tidak valid");
}

9. Perulangan (Loop)

For Loop:

for (let i = 1; i <= 5; i++) {
    console.log(i);
}
// Output: 1 2 3 4 5

While Loop:

let i = 1;
while (i <= 5) {
    console.log(i);
    i++;
}

Do-While Loop:

let i = 1;
do {
    console.log(i);
    i++;
} while (i <= 5);

For-Of (Array):

let buah = ["Apel", "Jeruk", "Mangga"];
for (let item of buah) {
    console.log(item);
}

For-In (Object):

let orang = {nama: "Budi", umur: 25};
for (let key in orang) {
    console.log(key + ": " + orang[key]);
}

Break & Continue:

for (let i = 1; i <= 10; i++) {
    if (i === 5) continue; // skip 5
    if (i === 8) break;    // stop di 8
    console.log(i);
}
// Output: 1 2 3 4 6 7

11. Array

Membuat Array:

let buah = ["Apel", "Jeruk", "Mangga"];
let angka = [1, 2, 3, 4, 5];
let campur = ["Teks", 123, true, null];

Akses Array:

console.log(buah[0]); // "Apel"
console.log(buah[1]); // "Jeruk"
console.log(buah.length); // 3

Array Methods:

let arr = [1, 2, 3];

// Push - tambah di akhir
arr.push(4); // [1, 2, 3, 4]

// Pop - hapus dari akhir
arr.pop(); // [1, 2, 3]

// Unshift - tambah di awal
arr.unshift(0); // [0, 1, 2, 3]

// Shift - hapus dari awal
arr.shift(); // [1, 2, 3]

// Splice - hapus/tambah di tengah
arr.splice(1, 1); // hapus index 1, [1, 3]
arr.splice(1, 0, 2); // tambah di index 1, [1, 2, 3]

// Slice - ambil sebagian array
let arr2 = arr.slice(0, 2); // [1, 2]

// Join - gabung jadi string
console.log(arr.join("-")); // "1-2-3"

// Concat - gabung array
let arr3 = arr.concat([4, 5]); // [1, 2, 3, 4, 5]

// IndexOf - cari index
console.log(arr.indexOf(2)); // 1

// Includes - cek ada/tidak
console.log(arr.includes(2)); // true

// Reverse - balik urutan
arr.reverse(); // [3, 2, 1]

// Sort - urutkan
arr.sort(); // [1, 2, 3]

Array Higher-Order Functions:

let angka = [1, 2, 3, 4, 5];

// Map - transform setiap element
let kali2 = angka.map(x => x * 2);
console.log(kali2); // [2, 4, 6, 8, 10]

// Filter - filter berdasarkan kondisi
let genap = angka.filter(x => x % 2 === 0);
console.log(genap); // [2, 4]

// Reduce - gabung jadi satu nilai
let total = angka.reduce((sum, x) => sum + x, 0);
console.log(total); // 15

// Find - cari element pertama yang cocok
let hasil = angka.find(x => x > 3);
console.log(hasil); // 4

// FindIndex - cari index pertama yang cocok
let index = angka.findIndex(x => x > 3);
console.log(index); // 3

// Some - cek apakah ada yang cocok
console.log(angka.some(x => x > 4)); // true

// Every - cek apakah semua cocok
console.log(angka.every(x => x > 0)); // true

// ForEach - loop tanpa return
angka.forEach(x => console.log(x));

console.log(gabung2); // {a:1, b:2, c:3, d:4}

Destructuring:

// Array destructuring

19. Fetch API (AJAX)

// GET request
fetch("https://api.example.com/data")
    .then(response => response.json())
    .then(data => {
        console.log(data);
    })
    .catch(error => {
        console.log("Error:", error);
    });

// POST request
fetch("https://api.example.com/data", {
    method: "POST",
    headers: {
        "Content-Type": "application/json"
    },
    body: JSON.stringify({nama: "Budi", umur: 25})
})
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.log(error));

// Async/Await (modern)
async function getData() {
    try {
        const response = await fetch("https://api.example.com/data");
        const data = await response.json();
        console.log(data);
    } catch (error) {
        console.log("Error:", error);
    }
}

21. Tips JavaScript

  • Gunakan const & let, hindari var
  • Gunakan === instead of ==
  • Gunakan arrow function untuk code yang lebih clean
  • Gunakan template literals untuk string interpolation
  • Gunakan array methods (map, filter, reduce) daripada loop
  • Selalu handle error dengan try-catch
  • Gunakan async/await untuk code yang lebih readable
  • Buat nama variabel & function yang descriptive
  • Test code di console browser (F12)
  • Pelajari DevTools untuk debugging

22. Contoh Program Sederhana

// To-Do List
let todos = [];

function addTodo(text) {
    todos.push({
        id: Date.now(),
        text: text,
        completed: false
    });
    renderTodos();
}

function toggleTodo(id) {
    const todo = todos.find(t => t.id === id);
    if (todo) {
        todo.completed = !todo.completed;
        renderTodos();
    }
}

function deleteTodo(id) {
    todos = todos.filter(t => t.id !== id);
    renderTodos();
}

function renderTodos() {
    const list = document.getElementById("todoList");
    list.innerHTML = todos.map(todo => `
        <li>
            <input type="checkbox" 
                   ${todo.completed ? 'checked' : ''} 
                   onclick="toggleTodo(${todo.id})">
            <span style="${todo.completed ? 'text-decoration: line-through' : ''}">
                ${todo.text}
            </span>
            <button onclick="deleteTodo(${todo.id})">Hapus</button>
        </li>
    `).join('');
}
